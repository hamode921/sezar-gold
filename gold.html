async function fetchGoldData() { try { // استخدام رابط مباشر وموثوق لجلب الأسعار const response = await fetch('https://api.binance.com/api/v3/ticker/price?symbol=XAUUSDT'); const ticker = await response.json(); const currentPrice = parseFloat(ticker.price); document.getElementById('main-price').innerText = `$${currentPrice.toFixed(2)}`; // جلب السيولة (Order Book) const depthRes = await fetch('https://api.binance.com/api/v3/depth?symbol=XAUUSDT&limit=10'); const data = await depthRes.json(); checkWhales(data.bids, data.asks); updateStatus(data.bids, data.asks); } catch(e) { logMsg("جاري محاولة الاتصال بالسيولة..."); } }
